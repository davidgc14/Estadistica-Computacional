Ejercicio 1

Aproximar mediante simulacin la probabilidad P(X + Y 1). Compara con el valor exacto que vale /4.

# simulacin
nsim <- 1000
set.seed(1)
x <- runif(nsim,-1,1)
y <- runif(nsim,-1,1)
suceso <- (x+y <= 1)

# aproximacin
mean(suceso)
## [1] 0.878
# error de estimacin
sd(suceso)/sqrt(nsim)
## [1] 0.01035486
# Valor real
pi/4
## [1] 0.7853982
# Estudio de convergencia
estim <- cumsum(suceso)/(1:nsim)
plot(1:nsim,estim,type='l',
                ylab='Aproximacin y lmites de error',
                xlab='Nmero de simulaciones',
                main=expression(P(X+Y<=1)),
                ylim=c(0,2))

# Errores de estimacin
estim.err<-sqrt(cumsum((suceso-estim)^2))/(1:nsim)

# Limites de error
z<-qnorm(0.025,lower.tail = FALSE)
lines(estim-z*estim.err,lty=3,lwd=2,col='blue')
lines(estim+z*estim.err,lty=3,lwd=2,col='blue')
abline(h=pi/4,lty=2,col=2)

Ejercicio 2

Una vez visualizados los problemas, el objetivo es aproximar las dos integrales anteriores utilizando simulacin. Para ello te sugiero que consideres 1000 simulaciones. Adems de la aproximacin obtenida calcula su error de estimacin y construye un grco que muestre la convergencia junto con los lmites de error.

Primera integral:

f1 <- function(x) dbeta(x, 2.5, 5)
curve(f1,0.2,0.4)

nsim <- 1000
# calculamos la aproximacin
set.seed(1)
x <- runif(nsim, 0.2,0.4) # x1,...,xn
f1x <- sapply(x, f1) # f1(x1),...f1(xn)
mx <- mean(f1x)*0.2 # Se multiplica por la amplitud del intervalo

# Valor real
integrate(f1,0.2,0.4)
## 0.4441656 with absolute error < 4.9e-15
# Grfica de convergencia
# valores estimados
f1x. <- sapply(x, f1)*0.2 # Multiplicamos por la amplitud del intervalo
estim <- cumsum(f1x.)/(1:nsim)
# errores de estimacin
estim.err <- sqrt(cumsum((f1x.-estim)^2))/(1:nsim)
plot(1:nsim,estim,type='l',
                ylab='Aproximacin y lmites de error',
                xlab='Nmero de simulaciones')
# Extremos de error
z <- qnorm(0.025,lower.tail = FALSE)
lines(estim - z*estim.err,col='blue',lwd=2,lty=3)
lines(estim + z*estim.err,col='blue',lwd=2,lty=3)
abline(h=mx, col=2)

Ejercicio 3

Calcula la media y la varianza de los nsim valores simulados de S_N. Estos nos dan aproximaciones de la media y la varianza exactas que calculamos antes de forma exacta.

nsim <- 5000 # nmero de simulaciones

# simulamos ahora nsim valores de S_N:
lam <- 17
mu <- 3.5
sig <- 1.1

S <- double(nsim) # para almacenar los valores simulados
set.seed(1)
for (i in 1:nsim) {
    n <- rpois(1,lam)
    if (n > 0) 
        S[i] <- sum(rlnorm(n,mu,sig))
}

# Aproximacin de la media
EX <- mean(S); EX
## [1] 1028.819
# Convergencia
estim <- cumsum(S)/(1:nsim)
plot(1:nsim,estim,type='l',
                ylab='Aproximacin y lmites de error',
                xlab='Nmero de simulaciones')
abline(h=EX,col=2)

# Aproximacin de la varianza
var(S)
## [1] 206025.3
